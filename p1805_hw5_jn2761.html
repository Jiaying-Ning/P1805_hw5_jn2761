df = 
  tibble(
    path = list.files("data"),
  ) %>% 
  mutate(
    path = str_c("data/", path),
    data = map(path,read.csv)) %>%
  unnest() %>% 
   mutate(path=gsub("data/","",path),
          path=gsub(".csv","",path)) %>%
   separate(path, c("control_arm","subject_id"), sep = "([_])") %>%
  relocate(subject_id,control_arm) %>%
   pivot_longer(
    week_1:week_8,
    names_to = "week", 
    names_prefix = "week_",
    values_to = "observation")
